-- Free captured villagers --
function UserFunction_FreeVillager(idActorGizmo, idSource)
	local idCaptive = Hydra.ActorsLinkedByActor(idActorGizmo)[1]

-- The villager will play a conversation, have it's vfx removed, and run toward the escape location.
	if Hydra.ActorIsValid(idCaptive) and Hydra.ActorIsAlive(idCaptive) then
		Hydra.ActorPlaySimpleConversation(S_CONVO_VILLAGER_FREE, idCaptive)
		Hydra.QActorMoveToLocationDynamicSpeed(idCaptive, idEscapeLoc, "IGC_NPC_Sprint_Fast", false)
		Hydra.ActorRemoveComplexEffect(idCaptive, tRopes[idCaptive])
		Hydra.ActorRemoveFromGroup(idCaptive, S_CAPTIVES_GROUP)
		Hydra.QActorFadeAndDelete(idCaptive, 1)
	end
	
-- Drain the summoner of some health every time a villager is rescued.	
	local nCurrentHP = Hydra.ActorGetHitPointsPercent(idSummoner)
	local nNewHP = nCurrentHP - 0.16
	Hydra.ActorSetHitPointsPercent(idSummoner, nNewHP)
end
