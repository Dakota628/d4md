function SpawnMaw(vPos)
	if not Is_Maw_Active then

		Is_Maw_Active = true

		if Hydra.ActorIsValid(Ref_Id_BossSpawn_MawSkirt) then
			Hydra.ActorDelete(Ref_Id_BossSpawn_MawSkirt);
		end
		
		if Hydra.ActorIsValid(Ref_Id_BossSpawn_MawPortal) then
			Hydra.ActorDelete(Ref_Id_BossSpawn_MawPortal);
		end
		
		local vMawPos = Hydra.PositionOffsetVertically(vPos,-0.075);
	
		Ref_Id_BossSpawn_MawPortal = Hydra.ActorSpawnActor(S_MAW_PORTAL_ACTOR, vMawPos, Ref_VMawFacing, false, 0, 0);
		Ref_Id_BossSpawn_MawSkirt = Hydra.ActorSpawnActor(S_MAW_SKIRT_ACTOR, vMawPos, Ref_VMawFacing, false, 0, 0);
	
		if Hydra.ActorIsValid(Ref_Id_BossSpawn_MawPortal) and Hydra.ActorIsValid(Ref_Id_BossSpawn_MawSkirt)then
			EnableMaw(Ref_Id_BossSpawn_MawPortal, Ref_Id_BossSpawn_MawSkirt);
		end
	end
end

function PlayMawClosing(idPortal, idSkirt)
	if Is_Maw_Active then
	
		-- Make sure we only Play Closing once
		Is_Maw_Active = false;

		-- Portal goes away
		if Hydra.ActorIsValid(idPortal) then
			-- Wait before killing portal, this is similar to the anim duration of the skirt escape anim
			Hydra.QActorWait(idPortal, 1.3);
			Hydra.QActorKill(idPortal);
		end
	
		-- Skirt goes away
		if Hydra.ActorIsValid(idSkirt) then
			Hydra.QActorAnim(idSkirt, POWER_MAW_ESCAPE);
			Hydra.QActorKill(idSkirt);
		end

	end
end
